    <script>
      let myData;

      fetch('https://www.dnd5eapi.co/api/spells')
        .then(response => response.json())
        .then(data => {
          const spells = data.results;
          let spellList = '';

          // Loop through the spells and format them as a list
          for (let i = 0; i < spells.length; i++) {
            spellList += `<li>${spells[i].name} - ${spells[i].index}</li>`;
          }

          // Display the formatted spell list
          document.getElementById('myDataContainer').innerHTML = `<ul>${spellList}</ul>`;
        })
        .catch(error => {
          console.error('Error fetching data:', error);
        });
    </script>
